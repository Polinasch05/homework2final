        "# пишем функцию по сбору информации\n",
        "def GetNews(url0):\n",
        "    page0 = requests.get(url0)\n",
        "    soup0 = BeautifulSoup(page0.text, features=\"html.parser\")\n",
        "\n",
        "    # Извлекаем информацию с использованием метода find и проверки наличия элементов\n",
        "    author = soup0.find('meta', {'name': 'author'})\n",
        "    author = author.attrs['content'] if author else 'N/A'\n",
        "\n",
        "    date_tag = soup0.find('meta', {'property': 'article:published_time'})\n",
        "    publication_date = date_tag['content'] if date_tag else 'Date not found'\n",
        "\n",
        "    title = soup0.find('meta', {'property': 'og:title'})\n",
        "    title = title.attrs['content'] if title else 'N/A'\n",
        "\n",
        "    final_text = \"\"\n",
        "    for i in soup0.find_all('p'):\n",
        "        final_text += i.text.strip() + '\\n'\n",
        "\n",
        "    return {'url': url0, 'author': author, 'date': publication_date, 'title': title, 'text': final_text}"
